# 上传图片到github 返回图片的 cdn链接

# 使用

# axios请求

import axios from "axios";


export const uploadToGithub = (filename, base64Img) => {
  const URL = `https://api.github.com/repos/hape666/cdn/contents/admin_img/${filename}?access_token=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`
  return axios({
    url: URL,
    method: 'put',
    headers: {
      Accept: "application/vnd.github.v3+json"
    },
    data: {
      "message": "upload_img",
      "committer": {
        "name": "hape666",
        "email": "peng11702@gmail.com"
      },
      "content": base64Img
    }
  })
}

# markdown.vue

<template>
  <div>
    <div class="fileBox">
      <input type="file" ref="file">
      <button @click="upload">文件上传</button>
      <p style="color: red">{{ msg }}</p>
      <img :src="img_url">
    </div>
    <mavon-editor v-model="md_value" ref="md" @imgAdd="imgAdd"/>
  </div>
</template>

<script>
import {uploadToGithub} from "../api/upload";
import getFileBase64 from "../utils/toBase64";

export default {
  name: "markdown",
  data() {
    return {
      md_value: '',
      msg: '',
      img_url_pre: 'http://cdn.jsdelivr.net/gh/hape666/cdn/',
      img_url: ''
    }
  },
  methods: {
    // 绑定@imgAdd event
    async imgAdd(pos, file) {

      console.log(pos, file)
      let base64Img = file.miniurl.split(',')[1]
      // console.log(base64Img)
      //获取当前时间戳
      let timestamp = new Date().getTime()
      let filename = timestamp + "-" + file.name
      console.log(filename)
      let res = await uploadToGithub(filename, base64Img)
      if (res) {
        this.img_url = this.img_url_pre + res.data.content.path
        this.$refs['md'].$img2Url(pos, this.img_url);
        console.log("上传成功")
      }
    },
    async upload() {
      this.msg = "正在上传"
      console.log(this.$refs.file.files[0])
      if (this.$refs.file.files[0].type.indexOf("image") !== -1) {
        let base64Img = await getFileBase64(this.$refs.file)
        //获取当前时间戳
        let timestamp = new Date().getTime()
        let filename = timestamp + "-" + this.$refs.file.files[0].name
        let res = await uploadToGithub(filename, base64Img)
        console.log(res)
        if (res) {
          this.img_url = this.img_url_pre + res.data.content.path
          this.msg = "上传成功"
        }
      } else {
        this.msg = "error!,格式只能是图片格式"
        this.$refs.file.value = ''
      }

    }
  }
}

</script>

<style scoped>
.fileBox {
  text-align: center;

}

# 将图片转换成base64格式

const getFileBase64 = (file) => {
  return new Promise((resolve, reject) => {
    let reader = new FileReader();
    reader.readAsDataURL(file.files[0]);
    reader.onload = (e) => {
      console.log(e.target.result)

      let res = e.target.result.split(',')[1]
      resolve(res)
    }
  })
}
export default getFileBase64
