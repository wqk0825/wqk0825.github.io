# Axios 的二次封装

import axios from 'axios';

// 创建了一个axios 实例
let service = axios.create({
  baseURL:'http://bb.shoujiduoduo.com/baby/bb.php'
});

// 设置请求的拦截
service.interceptors.request.use(config => {
  // 设置请求头  token
  config.headers = {
    token: 'abc'
  };
  return config;
});



// /baby/bb.php?type=getvideos&pagesize=30&collectid=29


export default service;
